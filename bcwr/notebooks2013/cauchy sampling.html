heights <- c(64, 68, 62, 61, 62, 62, 65, 61, 65, 65, 84, 76)

# function cauchyerrorpost contains log posterior of (mu, log sigma)

mycontour(cauchyerrorpost,  c(55, 75, -1.2, 3.5), heights,
          xlab="MU", ylab="LOG SIGMA",
          main="Cauchy and Normal Sampling Posteriors")

# contrast with the posterior of (mu, log sigma) with normal
# sampling

# define a new function normalerrorpost

normalerrorpost <- function(theta, data){
  logf = function(data, theta) 
      dnorm(data, theta[1], exp(theta[2]), log=TRUE)
  return(sum(logf(data, theta)))
}

# show contours of the normal error posterior in red

mycontour(normalerrorpost, c(55, 75, -1.2, 3.5), heights, 
          add=TRUE, col="red")

# add a legend

legend(70, 0, legend=c("Cauchy", "Normal"), lty=1, 
       col=c("black", "red"))
