modelString = "
model{
  for (i in 1:N){
    y[i] ~ dpois(lam[i] * e[i])
    lam[i] ~ dgamma(a, b)
  }
  a <- exp(log.a)
  b <- exp(log.a) / mu
  log.mu <- log(mu)
  mu ~ dgamma(0.0001, 0.0001)
  log.a ~ dlogis(-0.635, 1)
}"

writeLines(modelString, con="poissonexch.bug")

##################  MODEL 2 - normal priors on log rates

modelString = "
model{
for (i in 1:N){
   y[i] ~ dpois(lam[i] * e[i])
   log(lam[i]) <- theta[i]
   theta[i] ~ dnorm(mu, tau)
}
mu ~ dnorm(0,.00001)
tau ~ dgamma(0.2, 0.2)
}"

writeLines(modelString, con="poissonexch2.bug")
